function draw () { if (player.x >= 0) { if (keyIsDown(LEFT_ARROW)) { delta.left += 1; player.x -= player.speedXleft; if (delta.left >= 20) { delta.left = 0; player.speedXleft += 1.25; } } else { player.speedXleft = 3; } } if (player.x + player.l <= 1200) { if (keyIsDown(RIGHT_ARROW)) { delta.right += 1; player.x += player.speedXright; if (delta.right >= 20) { delta.right = 0; player.speedXright += 1.25; } } else { player.speedXright = 3; } } if (player.y >= 0) { if (keyIsDown(UP_ARROW)) { delta.up += 1; player.y -= player.speedYup; if (delta.up >= 20) { delta.up = 0; player.speedYup += 1.25; } } else { player.speedYup = 3; } } if (player.y + player.h <= 900) { if (keyIsDown(DOWN_ARROW)) { delta.down += 1; player.y += player.speedYdown; if (delta.down >= 20) { delta.down = 0; player.speedYdown += 1.25; } } else { player.speedYdown = 3; } }
